CREATE TABLE boss
(
    `usernum`     INT            NOT NULL    AUTO_INCREMENT, 
    `userid`      VARCHAR(45)    NOT NULL, 
    `userpw`      VARCHAR(45)    NOT NULL, 
    `userphone`   VARCHAR(45)    NOT NULL, 
    `username`    TEXT           NOT NULL, 
    `signuptime`  TIMESTAMP      NOT NULL, 
    PRIMARY KEY (usernum)
);

CREATE TABLE store
(
    `storenum`   INT              NOT NULL    AUTO_INCREMENT, 
    `storename`  VARCHAR(45)      NOT NULL, 
    `people`     INT              NOT NULL, 
    `kind`       INT              NOT NULL, 
    `mode`       VARCHAR(45)      NOT NULL, 
    `address`    VARCHAR(2000)    NOT NULL, 
    `opentime`   VARCHAR(25)      NOT NULL, 
    `closstime`  VARCHAR(25)      NOT NULL, 
    `phone`      text             NOT NULL, 
    PRIMARY KEY (storenum)
);

CREATE TABLE guest
(
    `usernum`     INT             NOT NULL    AUTO_INCREMENT, 
    `userid`      VARCHAR(45)     NOT NULL, 
    `userpw`      VARCHAR(45)     NOT NULL, 
    `name`        VARCHAR(45)     NOT NULL, 
    `age`         INT             NOT NULL, 
    `gender`      INT             NOT NULL, 
    `email`       VARCHAR(200)    NOT NULL, 
    `writing`     text            NULL, 
    `phone`       text            NOT NULL, 
    `signuptime`  TIMESTAMP       NOT NULL, 
    PRIMARY KEY (usernum)
);

CREATE TABLE menu
(
    `menunum`    INT             NOT NULL    AUTO_INCREMENT, 
    `storenum`   INT             NOT NULL, 
    `kindnum`    INT             NOT NULL, 
    `menuname`   VARCHAR(45)     NOT NULL, 
    `menuprice`  VARCHAR(100)    NOT NULL, 
    PRIMARY KEY (menunum)
);

CREATE TABLE review
(
    `reviewnum`   INT          NOT NULL    AUTO_INCREMENT, 
    `usernum`     INT          NOT NULL, 
    `storenum`    INT          NOT NULL, 
    `text`        TEXT         NOT NULL, 
    `star`        INT          NOT NULL, 
    `reviewtime`  TIMESTAMP    NOT NULL, 
    PRIMARY KEY (reviewnum)
);


CREATE TABLE bookmark
(
    `marknum`   INT    NOT NULL    AUTO_INCREMENT, 
    `usernum`   INT    NOT NULL, 
    `storenum`  INT    NOT NULL, 
    PRIMARY KEY (marknum)
);

CREATE TABLE mychachacha
(
    `chanum`    INT            NOT NULL    AUTO_INCREMENT, 
    `usernum`   VARCHAR(45)    NOT NULL, 
    `storenum`  VARCHAR(45)    NOT NULL, 
    `chaname`   TEXT           NOT NULL, 
    PRIMARY KEY (chanum)
);
